\makeatletter
\def\DeclareFigureOptions#1#2#3{%
    \if\relax\detokenize{#2}\relax
        \define@key{Gin}{figopts@\detokenize{#1}@twocol}[true]{}%
    \else
        \define@key{Gin}{figopts@\detokenize{#1}@twocol}[true]{\setkeys{Gin}{width=#2}}%
    \fi
    \if\relax\detokenize{#3}\relax
        \define@key{Gin}{figopts@\detokenize{#1}@onecol}[true]{}%
    \else
        \define@key{Gin}{figopts@\detokenize{#1}@onecol}[true]{\setkeys{Gin}{width=#3}}%
    \fi
}

\ifdefined\JOURNALTWOCOLUMN
    \def\FigureOptions#1{figopts@\detokenize{#1}@twocol}
\else
    \def\FigureOptions#1{figopts@\detokenize{#1}@onecol}
\fi
\makeatother
